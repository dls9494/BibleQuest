+      ..quadraticBezierTo(w * 0.20, h * 0.26, 0, h * 0.30)

+      ..moveTo(-w * 0.015, -h * 0.22)
+      ..quadraticBezierTo(-w * 0.20, -h * 0.26, -w * 0.40, -h * 0.20)
+      ..lineTo(-w * 0.40, h * 0.20)
+      ..quadraticBezierTo(-w * 0.20, h * 0.24, -w * 0.015, h * 0.22)

+      begin: Alignment.centerLeft,
+      end: Alignment.centerRight,

+      ..color = const Color(0xFF8B6C3F)
+      ..style = PaintingStyle.stroke

+    canvas.drawPath(rightPagePath, pageBorderPaint);

+      ..color = const Color(0xFF5E4221).withValues(alpha: 0.95)

+      ..style = PaintingStyle.stroke

+    void drawPageGlyph(String text, double cx, double cy, double gw, double gh) {
+      final len = text.length;
+      final totalW = gw * len * 1.1;
+      final step = len > 1 ? totalW / (len - 1) : 0.0;

+        canvas.translate(w * 0.005, h * 0.005);

+        canvas.restore();
+        _drawGlyph(canvas, charX, cy, gw, gh, text[i], textPaint);

+    drawPageGlyph("SRVT", -w * 0.20, -h * 0.04, charW, charH);

+      ..color = const Color(0xFFFFFDF5)
+      ..style = PaintingStyle.fill;

+      ..color = const Color(0xFFCFA83B)

+        topLeft: Radius.circular(w * 0.04),
+        topRight: Radius.circular(w * 0.04),
+      ));
+    canvas.drawPath(headerPath, headerPaint);

+    // Calendar Header Text (Calendar) - represented by a tiny horizontal line
+    final whiteLinePaint = Paint()
+      ..color = Colors.white
+      ..style = PaintingStyle.stroke
+      ..strokeWidth = w * 0.008
+      ..strokeCap = StrokeCap.round;
+    canvas.drawLine(Offset(-calW * 0.25, -calH * 0.36), Offset(calW * 0.25, -calH * 0.36), whiteLinePaint);
+
+    // Days of week S M T W T F S (represented by tiny vertical ticks or dots)
+    final tickPaint = Paint()
+      ..color = const Color(0xFF5E4221)
+      ..style = PaintingStyle.stroke
+      ..strokeWidth = w * 0.006
+      ..strokeCap = StrokeCap.round;

+    final tickStep = calW * 0.70 / 6;
+    final tickY = -calH * 0.08;
+    for (int i = 0; i < 7; i++) {
+      final tx = startTickX + i * tickStep;
+      canvas.drawLine(Offset(tx, tickY - h * 0.012), Offset(tx, tickY + h * 0.012), tickPaint);
+    }
+
+    // Calendar numbers grid (small dots for dates)
+    final gridDotPaint = Paint()
+      ..color = const Color(0xFF8B7355)
+      ..style = PaintingStyle.fill;
+    
+    final gridYStart = calH * 0.06;
+    final gridRowStep = calH * 0.18;
+    for (int row = 0; row < 3; row++) {
+      final gy = gridYStart + row * gridRowStep;
+      for (int col = 0; col < 7; col++) {
+        // Skip some positions to look like a real calendar month starting in mid-week
+        if (row == 0 && col < 3) continue; // Start on Wednesday
+        final gx = startTickX + col * tickStep;
+        
+        // Let's make one date (e.g. Wednesday in the 3rd row, which would be the 16th) circled in red!
+        if (row == 2 && col == 1) {
+          // Draw red circle around this date instead of just a dot!
+          final redCirclePaint = Paint()
+            ..color = const Color(0xFFE53935)
+            ..style = PaintingStyle.stroke
+            ..strokeWidth = w * 0.012;
+          canvas.drawCircle(Offset(gx, gy), w * 0.026, redCirclePaint);
+          // And a tiny dark dot inside the circle
+          canvas.drawCircle(Offset(gx, gy), w * 0.006, Paint()..color = const Color(0xFFE53935));
+        } else {
+          canvas.drawCircle(Offset(gx, gy), w * 0.008, gridDotPaint);
+        }
+      }
+    }
+
+    // Top metal rings binding the calendar
+    final ringPaint = Paint()
+      ..color = const Color(0xFFB0BEC5)
+      ..style = PaintingStyle.stroke
+      ..strokeWidth = w * 0.016
+      ..strokeCap = StrokeCap.round;
+    
+    void drawRing(double rx) {
+      final ringPath = Path()
+        ..moveTo(rx, -calH * 0.50)
+        ..quadraticBezierTo(rx, -calH * 0.68, rx + w * 0.02, -calH * 0.68)
+        ..quadraticBezierTo(rx + w * 0.04, -calH * 0.68, rx + w * 0.04, -calH * 0.50);
+      canvas.drawPath(ringPath, ringPaint);
+    }
+    drawRing(-calW * 0.22);
+    drawRing(calW * 0.12);
+
+    canvas.restore(); // Restore calendar coordinate system
+  }

+  void _drawGlyph(Canvas canvas, double cx, double cy, double gw, double gh, String glyph, Paint paint) {
+    final w2 = gw / 2;
+    final h2 = gh / 2;

+      case 'A': // Aleph
+        canvas.drawLine(Offset(cx - w2, cy - h2), Offset(cx + w2, cy + h2), paint);
+        canvas.drawLine(Offset(cx + w2 * 0.1, cy - h2 * 0.1), Offset(cx + w2, cy - h2 * 0.8), paint);
+        canvas.drawLine(Offset(cx - w2 * 0.1, cy + h2 * 0.1), Offset(cx - w2, cy + h2 * 0.8), paint);
+        break;
+      case 'Y': // Ayin
+        canvas.drawLine(Offset(cx - w2, cy - h2), Offset(cx, cy), paint);
+        canvas.drawLine(Offset(cx + w2, cy - h2), Offset(cx - w2 * 0.5, cy + h2), paint);
+        break;
+      case 'L': // Lamed
+        final path = Path()
+          ..moveTo(cx - w2 * 0.2, cy - h2 * 1.5)
+          ..quadraticBezierTo(cx - w2 * 0.6, cy - h2 * 1.5, cx - w2 * 0.4, cy - h2 * 0.8)
+          ..lineTo(cx + w2 * 0.2, cy)
+          ..lineTo(cx + w2 * 0.2, cy + h2 * 0.4)
+          ..quadraticBezierTo(cx + w2 * 0.2, cy + h2, cx - w2 * 0.8, cy + h2);
+        canvas.drawPath(path, paint);
+        break;
+      case 'B': // Bet
+        final path = Path()
+          ..moveTo(cx - w2 * 0.8, cy - h2)
+          ..lineTo(cx + w2 * 0.8, cy - h2)
+          ..lineTo(cx + w2 * 0.8, cy + h2)
+          ..lineTo(cx - w2 * 1.1, cy + h2);
+        canvas.drawPath(path, paint);
+        break;
+      case 'H': // He
+        canvas.drawLine(Offset(cx - w2 * 0.8, cy - h2), Offset(cx + w2 * 0.8, cy - h2), paint);
+        canvas.drawLine(Offset(cx + w2 * 0.8, cy - h2), Offset(cx + w2 * 0.8, cy + h2), paint);
+        canvas.drawLine(Offset(cx - w2 * 0.8, cy - h2 * 0.3), Offset(cx - w2 * 0.8, cy + h2), paint);
+        break;
+      case 'N': // Nun
+        final path = Path()
+          ..moveTo(cx - w2 * 0.3, cy - h2)
+          ..lineTo(cx + w2 * 0.3, cy - h2)
+          ..lineTo(cx + w2 * 0.3, cy + h2)
+          ..lineTo(cx - w2 * 0.6, cy + h2);
+        canvas.drawPath(path, paint);
+        break;
+      case 'S': // Shin
+        final path = Path()
+          ..moveTo(cx - w2 * 0.8, cy - h2)
+          ..quadraticBezierTo(cx - w2 * 0.8, cy + h2, cx, cy + h2)
+          ..quadraticBezierTo(cx + w2 * 0.8, cy + h2, cx + w2 * 0.8, cy - h2)
+          ..moveTo(cx, cy + h2)
+          ..lineTo(cx, cy - h2 * 0.3);
+        canvas.drawPath(path, paint);
+        break;
+      case 'R': // Resh
+        final path = Path()
+          ..moveTo(cx - w2 * 0.8, cy - h2)
+          ..lineTo(cx + w2 * 0.8, cy - h2)
+          ..lineTo(cx + w2 * 0.8, cy + h2);
+        canvas.drawPath(path, paint);
+        break;
+      case 'V': // Vav
+        final path = Path()
+          ..moveTo(cx - w2 * 0.2, cy - h2)
+          ..lineTo(cx + w2 * 0.2, cy - h2)
+          ..lineTo(cx + w2 * 0.2, cy + h2);

+          ..lineTo(cx - w2 * 0.8, cy + h2 * 0.6)
+          ..quadraticBezierTo(cx - w2 * 0.8, cy + h2, cx - w2 * 1.2, cy + h2);
+        canvas.drawPath(path, paint);
+        break;
+      case 'M': // Mem
+        final path = Path()
+          ..moveTo(cx - w2 * 0.4, cy + h2)
+          ..lineTo(cx + w2 * 0.8, cy + h2)
+          ..lineTo(cx + w2 * 0.8, cy - h2)
+          ..lineTo(cx - w2 * 0.8, cy - h2)
+          ..lineTo(cx - w2 * 0.8, cy + h2 * 0.4);

+        break;
+      case 'I': // Yod
+        final path = Path()
+          ..moveTo(cx - w2 * 0.3, cy - h2)
